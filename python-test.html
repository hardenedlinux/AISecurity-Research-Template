import textwrap
def wrap_text(text, width):
    lines = text.split('\n')
    wrapped_lines = []
    for line in lines:
        if len(line) > width:
            wrapped_lines.extend(textwrap.wrap(line, width=width))
        else:
            wrapped_lines.append(line)
    return '\n'.join(wrapped_lines)
